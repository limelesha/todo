GET task(id: int)

POST task(
    project_id: int,
    title: str,
    supertask_id: int | None = None,
    description: str | None = None,
    due_timestamp: str | None = None,
)

PATCH task(
    id: int,
    title: str | None = None,
    description: str | None = None,
    due_timestamp: str | None = None,
    is_done: bool | None = None
)

DELETE task(id: int)

GET project(id: int)  // info about a project with a list of tasks

POST project(
    title: str,
    defalt_access_level_int: int = 1  // defaults to READER
)

PATCH project(
    id: int,
    title: str | None = None
    defalt_access_level_int: int | None = None,
)

DELETE project(id: int)

GET user(id: int)  // info about a user with a list of project they have access to

POST user(
    name: str,
    email: str,
    cleartext_password: str
)

PATCH user(
    id: int,
    email: str | None = None,
    password: str | None = None
)

DELETE user(id: int)

POST invite(  // invite new members which are granted project's default access level
    project_id: int,
    members: list  // list of invited members' IDs
)

POST kick(
    project_id: int,
    members: list  // list of members to remove from project's team
)

POST leave(
    user_id: int,
    project_id: int
)
